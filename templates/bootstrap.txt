<?php
use Slim\App;

$config = (new $name\Module)->loadDependencies();
$app    = new App($config);

// DIC configuration
$container = $app->getContainer();

foreach ($container->get('modules') as $moduleNamespace) {
    $container->get($moduleNamespace.'\Init');
}

// Register middleware
require 'middleware.php';

// Register routes
require 'routes.php';

$app->run();
